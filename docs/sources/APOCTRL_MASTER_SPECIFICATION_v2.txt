2. SQL-DATENBANKSCHEMA
----------------------

CREATE TABLE users (
 id INTEGER PRIMARY KEY,
 username TEXT UNIQUE,
 password_hash TEXT,
 role TEXT,
 created_at DATETIME
);

CREATE TABLE products (
 id INTEGER PRIMARY KEY,
 pzn TEXT UNIQUE,
 name TEXT,
 manufacturer TEXT,
 cost_price REAL,
 sale_price REAL,
 min_price REAL,
 target_rank INTEGER,
 strategy TEXT,
 category TEXT,
 active INTEGER,
 created_at DATETIME,
 updated_at DATETIME
);

CREATE TABLE competitors (
 id INTEGER PRIMARY KEY,
 name TEXT,
 url TEXT,
 active INTEGER
);

CREATE TABLE price_snapshots (
 id INTEGER PRIMARY KEY,
 product_id INTEGER,
 competitor_id INTEGER,
 price REAL,
 shipping_cost REAL,
 ranking INTEGER,
 captured_at DATETIME
);

CREATE TABLE own_price_snapshots (
 id INTEGER PRIMARY KEY,
 product_id INTEGER,
 price REAL,
 captured_at DATETIME
);

CREATE TABLE suggestions (
 id INTEGER PRIMARY KEY,
 product_id INTEGER,
 current_price REAL,
 suggested_price REAL,
 reason TEXT,
 created_at DATETIME
);

CREATE TABLE alerts (
 id INTEGER PRIMARY KEY,
 product_id INTEGER,
 type TEXT,
 priority TEXT,
 message TEXT,
 status TEXT,
 created_at DATETIME
);

CREATE TABLE audit_logs (
 id INTEGER PRIMARY KEY,
 user_id INTEGER,
 action TEXT,
 details TEXT,
 created_at DATETIME
);

5. RANKING ENGINE
-----------------
Berechnung:
Endpreis = Preis + Versand

Regeln:
- günstigster Preis = Rang 1
- nicht lieferbare Anbieter ignorieren
- Preisgleichheit erlaubt
